create table Author
(AuthorId int primary key,
AuthorName nvarchar(50))

create table Publisher
(PublisherName nvarchar(50) primary key,
Address nvarchar(50))

create table BookCategory
(category nvarchar(50) primary key)

create table Book
(BookId int primary key,
BookName nvarchar(50),
Price float,
category nvarchar(50) foreign key references BookCategory,
PublisherName nvarchar(50) foreign key references Publisher)

create table BookAuthor
(BookId int foreign key references Book,
AuthorId int foreign key references Author)
